How to Use These Assets

1. Setting Up Your Game Grid

const TILE_SIZE = 32; // Size of each tile in pixels const GRID_WIDTH = 20; // Number of tiles horizontally const GRID_HEIGHT = 20; // Number of tiles vertically

2. Loading Assets


// Example using JavaScript and Canvas
const tileMap = {
  WALL: 0,
  FLOOR: 1,
  DOOR: 2
};

function drawTile(ctx, type, x, y) {
  const tileSize = 32;
  switch(type) {
    case tileMap.WALL:
      ctx.fillStyle = '#2d2d2d';
      ctx.fillRect(x * tileSize, y * tileSize, tileSize, tileSize);
      break;
    // ... other tile types
  }
}
